64  {

66  if (!SolverBase<Move>::selectNode(node, validMoves)) {

110 template<class Move>
111 class SOSolver<Move, RootParallel> : public SOSolverBase<Move>
112 {
113 public:
114  using SOSolverBase<Move>::SOSolverBase;
115 
116  explicit SOSolver(std::size_t iterationCount = 1000, double exploration = 0.7)
117  : SOSolverBase<Move>{iterationCount / std::thread::hardware_concurrency(), exploration}
118  {}
119 
120  virtual Move operator()(const Game<Move> &rootState) const override
121  {
122  const auto numThreads = std::thread::hardware_concurrency();
123  std::vector<std::thread> threads(numThreads);
124  std::vector<Node<Move>> trees(numThreads);
125 
126  for (std::size_t t = 0; t < numThreads; ++t)
127  threads[t] = std::thread(&SOSolver::iterate, this, std::ref(trees[t]), std::ref(rootState));
128  for (auto &t : threads)
129  t.join();
130 
131  const auto results = compileVisitCounts(trees);
132  using pair = typename VisitMap::value_type;
133  const auto &mostVisited = *max_element(results.begin(), results.end(), [](const pair &a, const pair &b){
134  return a.second < b.second;
135  });
136  return mostVisited.first;
137  }
138 
139 private:
140  using VisitMap = std::map<Move, unsigned int>;
141 
142  // Map each unique move to its total number of visits
143  static VisitMap compileVisitCounts(const std::vector<Node<Move>> &trees)
144  {
145  VisitMap results;
146  for (auto &node : trees.front().children())
147  results.emplace(node->move(), node->visits());
148  for (auto t = trees.begin() + 1; t < trees.end(); ++t) {
149  for (auto &node : t->children()) {
150  const auto result = results.emplace(node->move(), node->visits());
151  if (!result.second)
152  (*result.first).second += node->visits();
153  }
154  }
155  return results;
156  }
157 };
