41  void iterate(RootList &trees, const Game<Move> &state) const
42  {
43  if (this->m_iterCount > 0) {
44  for (std::size_t i {0}; i < this->m_iterCount; ++i)
45  search(trees, state);
46  } else {
47  auto duration = this->m_iterTime;
48  while (duration.count() > 0) {
49  using namespace std::chrono;
50  const auto start = high_resolution_clock::now();
51  search(trees, state);
52  duration -= high_resolution_clock::now() - start;
53  }
54  }

58  void search(RootList &trees, const Game<Move> &rootState) const
59  {
60  NodePtrList roots(trees.size());
61  std::transform(trees.begin(), trees.end(), roots.begin(), [](Node<Move> &n){ return &n; });
62  auto randomState = rootState.cloneAndRandomise(rootState.currentPlayer());
63  auto statePtr = randomState.get();
64  select(roots, statePtr);
65  expand(roots, statePtr);
66  SolverBase<Move>::simulate(statePtr);
67  backPropagate(roots, statePtr);

76  void select(NodePtrList &nodes, Game<Move> *state) const
77  {
78  const auto validMoves = state->validMoves();
79  const auto player = state->currentPlayer();
80  const auto &targetNode = nodes[player];
81  if (!SolverBase<Move>::selectNode(targetNode, validMoves)) {
82  const auto selection = targetNode->ucbSelectChild(validMoves, this->m_exploration);
83  for (auto &node : nodes)
84  node = node->findOrAddChild(selection->move(), player);
85  state->doMove(selection->move());
86  select(nodes, state);
87  }
88  }

96  static void expand(NodePtrList &nodes, Game<Move> *state)
97  {
98  const auto player = state->currentPlayer();
99  const auto untriedMoves = nodes[player]->untriedMoves(state->validMoves());
100  if (!untriedMoves.empty()) {
101  const auto move = SolverBase<Move>::randMove(untriedMoves);
102  for (auto &node : nodes)
103  node = node->findOrAddChild(move, player);
104  state->doMove(move);
105  }
106  }

108  static void backPropagate(NodePtrList &nodes, const Game<Move> *state)
109  {
110  for (auto node : nodes)
111  SolverBase<Move>::backPropagate(node, state);
112  }

142 template<class Move>
143 class MOSolver<Move, RootParallel> : public MOSolverBase<Move>
144 {
145 public:
146  using MOSolverBase<Move>::MOSolverBase;
147 
148  explicit MOSolver(std::size_t numPlayers, std::size_t iterMax = 1000, double exploration = 0.7)
149  : MOSolverBase<Move>{numPlayers, iterMax / std::thread::hardware_concurrency(), exploration}
150  {}
151 
152  virtual Move operator()(const Game<Move> &rootState) const override
153  {
154  const auto numThreads = std::thread::hardware_concurrency();
155  std::vector<std::thread> threads(numThreads);
156  std::vector<RootList> treeSets(numThreads);
157  for (auto &set : treeSets)
158  set = RootList(this->m_numPlayers);
159 
160  for (std::size_t t = 0; t < numThreads; ++t)
161  threads[t] = std::thread(&MOSolver::iterate, this, std::ref(treeSets[t]), std::ref(rootState));
162  for (auto &t : threads)
163  t.join();
164 
165  // Gather results for the current player from each thread
166  NodePtrList currentPlayerTrees(numThreads);
167  std::transform(treeSets.begin(), treeSets.end(), currentPlayerTrees.begin(), [&](RootList &set){
168  return &set.at(rootState.currentPlayer());
169  });
170  const auto results = compileVisitCounts(currentPlayerTrees);
171  using pair = typename VisitMap::value_type;
172  const auto &mostVisited = *max_element(results.begin(), results.end(), [](const pair &a, const pair &b){
173  return a.second < b.second;
174  });
175  return mostVisited.first;
176  }
177 
178 protected:
179  using typename MOSolverBase<Move>::RootList;
180  using typename MOSolverBase<Move>::NodePtrList;
181 
182 private:
183  using VisitMap = std::map<Move, unsigned int>;
184 
185  // Map each unique move to its total number of visits
186  static VisitMap compileVisitCounts(const NodePtrList &trees)
187  {
188  VisitMap results;
189  for (auto &node : trees.front()->children())
190  results.emplace(node->move(), node->visits());
191  for (auto t = trees.begin() + 1; t < trees.end(); ++t) {
192  for (auto &node : (*t)->children()) {
193  const auto result = results.emplace(node->move(), node->visits());
194  if (!result.second)
195  (*result.first).second += node->visits();
196  }
197  }
198  return results;
199  }
200 };
