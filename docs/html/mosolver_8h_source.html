60  void iterate(RootList &trees, const Game<Move> &state) const
61  {
62  const auto iterations = this->iterationCount();
63  if (iterations > 0) {
64  for (std::size_t i {0}; i < iterations; ++i)
65  search(trees, state);
66  } else {
67  auto duration = this->iterationTime();
68  while (duration.count() > 0) {
69  using namespace std::chrono;
70  const auto start = high_resolution_clock::now();
71  search(trees, state);
72  duration -= high_resolution_clock::now() - start;
73  }
74  }

78  void search(RootList &trees, const Game<Move> &rootState) const
79  {
80  NodePtrList roots(trees.size());
81  std::transform(trees.begin(), trees.end(), roots.begin(), [](Node<Move> &n){ return &n; });
82  auto randomState = rootState.cloneAndRandomise(rootState.currentPlayer());
83  select(roots, *randomState);
84  expand(roots, *randomState);
85  this->simulate(*randomState);

95  void select(NodePtrList &nodes, Game<Move> &state) const
96  {
97  const auto validMoves = state.validMoves();
98  const auto player = state.currentPlayer();
99  const auto &targetNode = nodes[player];
100  if (!SolverBase<Move>::selectNode(targetNode, validMoves)) {
101  const auto selection = targetNode->ucbSelectChild(validMoves, this->m_exploration);
102  for (auto &node : nodes)
103  node = node->findOrAddChild(selection->move(), player);
104  state.doMove(selection->move());
105  select(nodes, state);
106  }
107  }

115  void expand(NodePtrList &nodes, Game<Move> &state) const
116  {
117  const auto player = state.currentPlayer();
118  const auto untriedMoves = nodes[player]->untriedMoves(state.validMoves());
119  if (!untriedMoves.empty()) {
120  const auto move = this->randomMove(untriedMoves);
121  for (auto &node : nodes)
122  node = node->findOrAddChild(move, player);
123  state.doMove(move);
124  }

169  threads[t] = std::thread(&MOSolver::iterate, this, std::ref(treeSets[t]), std::ref(rootState));
170  for (auto &t : threads)
171  t.join();
172 
173  // Gather results for the current player from each thread
174  RootList currentPlayerTrees(numThreads());
175  std::transform(treeSets.begin(), treeSets.end(), currentPlayerTrees.begin(), [&](RootList &set){
176  return std::move(set[rootState.currentPlayer()]);
