63  {

76  void iterate(Node<Move> &root, const Game<Move> &state) const
77  {
78  const auto iterations = this->iterationCount();
79  if (iterations > 0) {
80  for (std::size_t i {0}; i < iterations; ++i)
81  search(&root, state);
82  } else {
83  auto duration = this->iterationTime();
84  while (duration.count() > 0) {
85  using namespace std::chrono;
86  const auto start = high_resolution_clock::now();
87  search(&root, state);
88  duration -= high_resolution_clock::now() - start;
89  }
90  }
